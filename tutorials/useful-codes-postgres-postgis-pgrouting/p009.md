# Random values

```sql
UPDATE tb_sezioni_d0
SET 
  rnd_positive = CASE
      WHEN (random() * 1 + 0)::integer = 1 THEN true
        ELSE false
        END
--
UPDATE tb_sezioni_d0
SET 
  rnd_value = random() * 1 + 0
--
SELECT 
sezioni_token,
CASE
  WHEN rnd_positive=true THEN c01 + (c01 * rnd_value/5)
  ELSE c01 - (c01 * rnd_value/5)
END AS c01_new,
c01,
c01 * rnd_value/5,
rnd_positive, rnd_value
FROM tb_sezioni_d0
LIMIT 100;
```
